        // Initialize with proper grid positioning

        function createPeriodicTable() {

            const table = d3.select('#periodicTable');

            

            // Create 9 rows × 18 columns grid

            for (let row = 1; row <= 9; row++) {

                for (let col = 1; col <= 18; col++) {

                    const element = elements.find(e => e.x === col && e.y === row);

                    

                    const cell = table.append('div')

                        .attr('class', 'element')

                        .style('grid-column', col)

                        .style('grid-row', row)

                        .html(element ? `

                            <div class="number">${element.number}</div>

                            <div class="symbol">${element.symbol}</div>

                        ` : '');

                    

                    if (element) {

                        cell.on('mouseover', () => showElementInfo(element))

                            .on('mouseout', () => d3.select('#infoPanel').style('display', 'none'));

                    }

                }

            }

        }
